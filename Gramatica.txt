Game ::= SetupPhase "start" BattlePhase
 
SetupPhase ::= SetupPhase SetupTurn ";"
             | <vazio>
 
SetupTurn ::= player ":" ShipList
 
ShipList ::= ShipList "," Ship
           | Ship
 
Ship ::= "ship" n_size "at" coord DIR
 
BattlePhase ::= BattlePhase BattleAction ";"
              | <vazio>
 
BattleAction ::= player ":" "fire" coord
 
player ::= "A" | "B"
 
n_size ::= NUM
 
coord ::= LETTER NUM
 
DIR ::= "vertical"
      | "v"
      | "horizontal"
      | "h"
